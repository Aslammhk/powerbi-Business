# Create Procfile for Railway (backend)
@"
web: uvicorn app.main:app --host 0.0.0.0 --port $PORT
worker: celery -A app.workers.celery_app worker --loglevel=info
"@ | Out-File -FilePath "backend\Procfile" -Encoding UTF8
Write-Host "✅ backend/Procfile created" -ForegroundColor Green


# Create nixpacks.toml for Railway (backend)
@"
[phases.setup]
nixPkgs = ["python311", "gcc", "libpq"]

[phases.install]
cmds = [
    "pip install -r requirements.txt",
    "playwright install chromium",
    "playwright install-deps chromium"
]

[start]
cmd = "uvicorn app.main:app --host 0.0.0.0 --port $PORT"
"@ | Out-File -FilePath "backend\nixpacks.toml" -Encoding UTF8
Write-Host "✅ backend/nixpacks.toml created" -ForegroundColor Green





# Create railway.json for backend
@"
{
  "$schema": "https://railway.app/railway.schema.json",
  "build": {
    "builder": "NIXPACKS"
  },
  "deploy": {
    "startCommand": "uvicorn app.main:app --host 0.0.0.0 --port $PORT",
    "restartPolicyType": "ON_FAILURE",
    "restartPolicyMaxRetries": 10
  }
}
"@ | Out-File -FilePath "backend\railway.json" -Encoding UTF8
